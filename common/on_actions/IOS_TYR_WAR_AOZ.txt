on_actions = {
	on_daily_TYR = {
		effect = {
			if = {
				limit = {
					has_country_flag = TYR_AOZ_FW
					has_war_with = AOZ
					casualties > 4250
				}
				white_peace = AOZ
				country_event = TYR.200
				clr_country_flag = TYR_AOZ_FW
			}
		}
	}
}

conversion_tanks = {
	add_modern_armor_to_army = {

		icon = generic_civil_support
		visible = {
			has_tech = post_war_modern_tank
		}
		available = {
			has_tech = post_war_modern_tank
		}
		fire_only_once = no
		
		cost = 25

		days_remove = 7
		complete_effect = {
			hidden_effect = {
				set_technology = {
					mbt_in_batallions = 1
					popup = no
				}	
			}
			custom_effect_tooltip = add_modern_armor_to_army_tt
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				has_equipment = {
					modern_tank_eq > 100
				}
				factor = 5
			}
		}
	}
	converse_mt_to_mbt = {

		icon = generic_civil_support

		visible = {
			OR = {
				has_country_flag = {
					flag = TC_num_MTMBT
					value < 11
				}
				NOT = {
					has_country_flag = {
						flag = TC_num_MTMBT
					}	
				}
			}
		}
		available = {
			has_equipment = {
				med_tank_eq > 49
			}
			custom_trigger_tooltip = {
				tooltip = converse_tanks_tt
				OR = {
					has_country_flag = {
						flag = TC_num_MTMBT
						value < 11
					}
					NOT = {
						has_country_flag = {
							flag = TC_num_MTMBT
						}	
					}
				}
			}
		}
		fire_only_once = no
		
		cost = 25

		days_remove = 40
		
		modifier = {
			civilian_factory_use = 4
		}
		complete_effect = {
			add_equipment_to_stockpile = {
				type = modern_tank_eq
				amount = 50
			}
			hidden_effect = {
				add_equipment_to_stockpile = {
					type = med_tank_eq
					amount = -50
				}
				if = {
					limit = {
						NOT = {
							has_country_flag = TC_num_MTMBT
						}
					}
					set_country_flag = {
						flag = TC_num_MTMBT
						value = 1
					}
				}
				else = {
					modify_country_flag = {
						flag = TC_num_MTMBT
						value = 1
					}
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}
}
